# Aplicao do Modelo DBScan e Clculo do ndice de Silhueta
# Atende ao requisito de "Mensurar a qualidade de modelos de clusterizao": Uso do ndice de silhueta com DBScan
from sklearn.cluster import DBSCAN

dbscan = DBSCAN(eps=0.5, min_samples=5).fit(df_scaled)
# Verificao para garantir que existem clusters antes de calcular o ndice de silhueta
if len(set(dbscan.labels_)) > 1:
    silhouette_dbscan = silhouette_score(df_scaled, dbscan.labels_)
    print("ndice de Silhueta para DBScan:", silhouette_dbscan)
else:
    print("DBScan no formou clusters distintos com os parmetros dados.")

# Verificando se o DBScan formou clusters
unique_labels_dbscan = set(dbscan.labels_)
if -1 in unique_labels_dbscan:
    unique_labels_dbscan.remove(-1)  # Removendo o label -1, que indica 'rudo'

if len(unique_labels_dbscan) > 0:
    # Adicionando etiquetas de cluster do DBScan ao DataFrame original (limpo)
    df_clean['cluster_label_dbscan'] = dbscan.labels_

    # Calculando as mdias por cluster
    cluster_means_dbscan = df_clean[df_clean['cluster_label_dbscan'] != -1].groupby('cluster_label_dbscan').mean()

    # Preparando um resumo descritivo para cada cluster
    for cluster in cluster_means_dbscan.index:
        print(f"\nCluster {cluster} (DBScan):")
        print(f"- Mdia de vendas de Handgun: {cluster_means_dbscan.loc[cluster, 'handgun']:.2f}")
        print(f"- Mdia de vendas de Long Gun: {cluster_means_dbscan.loc[cluster, 'long_gun']:.2f}")
        print(f"- Mdia de vendas de Other: {cluster_means_dbscan.loc[cluster, 'other']:.2f}")
        print(f"- Mdia total de verificaes de antecedentes: {cluster_means_dbscan.loc[cluster, 'totals']:.2f}")
else:
    print("DBScan no formou clusters distintos alm do rudo.")
